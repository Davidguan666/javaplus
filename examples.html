Beginner Examples

🎮 Pong

Beginner Game

The classic Pong game - perfect for learning the basics.

import std.graphics;
import std.input;

let ballX = 400.0;
let ballY = 300.0;
let ballDX = 5.0;
let ballDY = 3.0;
let paddleY = 250.0;
let score = 0;

start() {
    let window = Window.create("Pong", 800, 600);
    window.setFramerateLimit(60);
    
    while (window.isOpen()) {
        window.pollEvents();
        
        if (Input.isKeyPressed(Key.ESCAPE)) {
            window.close();
        }
        
        // Paddle movement
        if (Input.isKeyDown(Key.UP) && paddleY > 0) {
            paddleY = paddleY - 5;
        }
        if (Input.isKeyDown(Key.DOWN) && paddleY < 500) {
            paddleY = paddleY + 5;
        }
        
        // Ball physics
        ballX = ballX + ballDX;
        ballY = ballY + ballDY;
        
        if (ballY <= 0 || ballY >= 580) {
            ballDY = -ballDY;
        }
        if (ballX >= 780) {
            ballDX = -ballDX;
        }
        
        // Paddle collision
        if (ballX <= 30 && ballY >= paddleY && ballY <= paddleY + 100) {
            ballDX = -ballDX;
            score = score + 1;
        }
        
        if (ballX < 0) {
            ballX = 400;
            ballY = 300;
            score = 0;
        }
        
        // Draw
        window.clear(Color.BLACK);
        window.drawRect(10, paddleY, 20, 100, Color.WHITE);
        window.drawCircle(ballX, ballY, 10, Color.ORANGE);
        window.display();
    }
}

🟡 Bouncing Balls

Beginner Graphics

Multiple bouncing balls with physics.

import std.graphics;
import std.math;

// Ball array
let balls = [];

// Create ball function
func createBall(x, y) {
    return {
        x: x,
        y: y,
        dx: Math.random() * 4 - 2,
        dy: Math.random() * 4 - 2,
        radius: Math.random() * 15 + 5,
        color: Color.random()
    };
}

start() {
    let window = Window.create("Bouncing Balls", 800, 600);
    
    // Create initial balls
    for (let i = 0; i < 10; i++) {
        balls.push(createBall(400, 300));
    }
    
    while (window.isOpen()) {
        window.pollEvents();
        
        // Update balls
        for (let ball in balls) {
            ball.x = ball.x + ball.dx;
            ball.y = ball.y + ball.dy;
            
            // Bounce
            if (ball.x <= ball.radius || ball.x >= 800 - ball.radius) {
                ball.dx = -ball.dx;
            }
            if (ball.y <= ball.radius || ball.y >= 600 - ball.radius) {
                ball.dy = -ball.dy;
            }
        }
        
        // Draw
        window.clear(Color.BLACK);
        for (let ball in balls) {
            window.drawCircle(ball.x, ball.y, ball.radius, ball.color);
        }
        window.display();
    }
}

Intermediate Examples

🐍 Snake Game

Intermediate Game

Classic Snake game with growing tail and collision.

import std.graphics;
import std.input;
import std.time;

let snake = [{x: 10, y: 10}];
let direction = {x: 1, y: 0};
let food = {x: 15, y: 15};
let gridSize = 20;
let cellSize = 25;

start() {
    let window = Window.create("Snake", 500, 500);
    window.setFramerateLimit(10);
    
    while (window.isOpen()) {
        window.pollEvents();
        
        // Input
        if (Input.isKeyPressed(Key.UP) && direction.y == 0) {
            direction = {x: 0, y: -1};
        }
        if (Input.isKeyPressed(Key.DOWN) && direction.y == 0) {
            direction = {x: 0, y: 1};
        }
        if (Input.isKeyPressed(Key.LEFT) && direction.x == 0) {
            direction = {x: -1, y: 0};
        }
        if (Input.isKeyPressed(Key.RIGHT) && direction.x == 0) {
            direction = {x: 1, y: 0};
        }
        
        // Move snake
        let head = snake[0];
        let newHead = {
            x: head.x + direction.x,
            y: head.y + direction.y
        };
        
        // Check wall collision
        if (newHead.x < 0 || newHead.x >= gridSize || 
            newHead.y < 0 || newHead.y >= gridSize) {
            window.close();
        }
        
        snake.insert(0, newHead);
        
        // Check food
        if (newHead.x == food.x && newHead.y == food.y) {
            food = {
                x: Math.randomInt(0, gridSize - 1),
                y: Math.randomInt(0, gridSize - 1)
            };
        } else {
            snake.pop();
        }
        
        // Draw
        window.clear(Color.BLACK);
        
        // Draw food
        window.drawRect(
            food.x * cellSize, 
            food.y * cellSize, 
            cellSize - 2, 
            cellSize - 2, 
            Color.RED
        );
        
        // Draw snake
        for (let segment in snake) {
            window.drawRect(
                segment.x * cellSize,
                segment.y * cellSize,
                cellSize - 2,
                cellSize - 2,
                Color.GREEN
            );
        }
        
        window.display();
    }
}

Advanced Examples

🚀 Space Shooter

Advanced Game

Complete space shooter with entities and collision.

import std.graphics;
import std.input;

// Entity system
entity Bullet {
    x, y, speed;
    
    update() {
        y = y - speed;
    }
    
    render(window) {
        window.drawRect(x, y, 4, 10, Color.YELLOW);
    }
}

entity Enemy {
    x, y, speed;
    
    update() {
        y = y + speed;
        x = x + Math.sin(y / 50) * 2;
    }
    
    render(window) {
        window.drawRect(x, y, 30, 30, Color.RED);
    }
}

entity Player {
    x = 400.0;
    y = 550.0;
    speed = 5.0;
    
    update() {
        if (Input.isKeyDown(Key.LEFT) && x > 0) {
            x = x - speed;
        }
        if (Input.isKeyDown(Key.RIGHT) && x < 750) {
            x = x + speed;
        }
    }
    
    render(window) {
        window.drawRect(x, y, 50, 30, Color.CYAN);
    }
}

// Game state
let player = Player.create();
let bullets = [];
let enemies = [];
let score = 0;
let spawnTimer = 0;

start() {
    let window = Window.create("Space Shooter", 800, 600);
    window.setFramerateLimit(60);
    
    while (window.isOpen()) {
        window.pollEvents();
        
        if (Input.isKeyPressed(Key.ESCAPE)) {
            window.close();
        }
        
        // Shoot
        if (Input.isKeyPressed(Key.SPACE)) {
            bullets.push(Bullet.create({
                x: player.x + 23,
                y: player.y,
                speed: 8
            }));
        }
        
        // Spawn enemies
        spawnTimer = spawnTimer + 1;
        if (spawnTimer > 60) {
            enemies.push(Enemy.create({
                x: Math.randomInt(50, 750),
                y: -30,
                speed: 2 + Math.random() * 2
            }));
            spawnTimer = 0;
        }
        
        // Update
        player.update();
        
        for (let bullet in bullets) {
            bullet.update();
        }
        
        for (let enemy in enemies) {
            enemy.update();
        }
        
        // Remove off-screen
        bullets = bullets.filter(b => b.y > -10);
        enemies = enemies.filter(e => e.y < 650);
        
        // Collision
        for (let enemy in enemies) {
            for (let bullet in bullets) {
                if (bullet.x > enemy.x && bullet.x < enemy.x + 30 &&
                    bullet.y > enemy.y && bullet.y < enemy.y + 30) {
                    score = score + 100;
                    enemy.destroy();
                    bullet.destroy();
                }
            }
        }
        
        // Draw
        window.clear(Color.BLACK);
        player.render(window);
        
        for (let bullet in bullets) {
            bullet.render(window);
        }
        
        for (let enemy in enemies) {
            enemy.render(window);
        }
        
        window.display();
    }
}
